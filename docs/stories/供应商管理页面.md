## 偏离说明（仅此四处，其余照抄）

| openhanako | Lumen | 理由 |
|-----------|-------|------|
| `added-models.yaml`（YAML 文件） | `~/.lumen/config.json["providers"]`（JSON） | Lumen 已用 JSON 作配置格式，统一存储 |
| `engine.providerRegistry`（Engine 实例属性） | `get_provider_registry()` 模块级单例 | Python 没有 Engine 对象，用函数获取单例 |
| OAuth 全部逻辑 | 删除 | Lumen 不支持 OAuth |
| `YAML.dump` 原子写入 | `json.dumps` + tmp→rename 原子写入 | JSON 替代 YAML，原子写入逻辑保留 |

from urllib.parse import urlparse

KNOWN_AUTH_TYPES = {"api-key", "oauth", "none", "optional"}
_ALLOWS_MISSING = {"none", "optional"}

def normalize_auth_type(auth_type: str) -> str:
    return auth_type if auth_type in KNOWN_AUTH_TYPES else "api-key"

def auth_type_allows_missing_key(auth_type: str) -> bool:
    return normalize_auth_type(auth_type) in _ALLOWS_MISSING

def is_local_base_url(url: str) -> bool:
    try:
        h = urlparse(url).hostname or ""
        return h in ("localhost", "127.0.0.1", "::1")
    except Exception:
        return False

def allows_missing_api_key(auth_type: str, base_url: str = "") -> bool:
    return auth_type_allows_missing_key(auth_type) or is_local_base_url(base_url)

OFFICIAL_DEEPSEEK_ID = "deepseek"
OFFICIAL_DEEPSEEK_HOST = "api.deepseek.com"
DEEPSEEK_RESERVED = {"deepseek"}

def _model_id_of(entry) -> str:
    if isinstance(entry, dict): return str(entry.get("id", ""))
    return str(entry) if isinstance(entry, str) else ""

def _hostname_of(base_url: str) -> str:
    try:
        from urllib.parse import urlparse
        return urlparse(base_url).hostname or ""
    except Exception:
        return ""

def _is_official_deepseek(provider_id: str, base_url: str = "") -> bool:
    return provider_id.strip().lower() == OFFICIAL_DEEPSEEK_ID \
        or _hostname_of(base_url).lower() == OFFICIAL_DEEPSEEK_HOST

def filter_discovered_models(provider_id: str, models: list, base_url: str = "") -> tuple[list, list]:
    """返回 (filtered, ignored_ids)"""
    filtered, ignored = [], []
    for m in models:
        mid = _model_id_of(m).strip()
        if _is_official_deepseek(provider_id, base_url) and mid.lower() in DEEPSEEK_RESERVED:
            if mid and mid not in ignored: ignored.append(mid)
            continue
        filtered.append(m)
    return filtered, ignored

def validate_provider_models(provider_id: str, models: list, base_url: str = "") -> None:
    _, ignored = filter_discovered_models(provider_id, models, base_url)
    if ignored:
        raise ValueError(
            f'Invalid model id "{ignored[0]}" for provider "{provider_id}": '
            f'"{ignored[0]}" is a provider id, not a model id.'
        )
```

import httpx

def build_auth_headers(api: str, api_key: str, allow_missing: bool = False) -> dict:
    """照抄 buildProviderAuthHeaders"""
    if not api:
        raise ValueError("api protocol is required")
    if not api_key and not allow_missing:
        raise ValueError("api_key is required")

    if api == "anthropic-messages":
        h = {"Content-Type": "application/json", "anthropic-version": "2023-06-01"}
        if api_key: h["x-api-key"] = api_key
        return h

    if api in ("openai-completions", "openai-responses"):
        h = {"Content-Type": "application/json"}
        if api_key: h["Authorization"] = f"Bearer {api_key}"
        return h

    if api == "google-generative-ai":
        h = {"Content-Type": "application/json"}
        if api_key: h["x-goog-api-key"] = api_key
        return h

    raise ValueError(f"Unsupported api protocol: {api}")

def build_probe_url(base_url: str, api: str) -> tuple[str, str]:
    """照抄 buildProbeUrl，返回 (url, method)"""
    base = base_url.rstrip("/")
    if api == "anthropic-messages":
        return f"{base}/v1/messages", "POST"
    return f"{base}/models", "GET"

async def probe_provider(base_url: str, api: str, api_key: str = "") -> dict:
    """照抄 probeProvider"""
    url, method = build_probe_url(base_url, api)
    headers = build_auth_headers(api, api_key, allow_missing=True) if api else {"Content-Type": "application/json"}

    body = None
    if method == "POST" and api == "anthropic-messages":
        import json
        body = json.dumps({"model": "test", "max_tokens": 1, "messages": [{"role": "user", "content": "."}]})

    try:
        async with httpx.AsyncClient(timeout=10) as client:
            resp = await client.request(method, url, headers=headers, content=body)
        auth_ok = resp.status_code not in (401, 403)
        return {"ok": auth_ok, "status": resp.status_code}
    except Exception as e:
        return {"ok": False, "status": 0, "error": str(e)}
```

## lib/providers/registry.py

**照抄 `provider-registry.js` 的所有非 OAuth 逻辑**，JSON 替换 YAML，去掉 mtime 缓存（JSON 读取够快）。

```python
"""ProviderRegistry — 照抄 openhanako core/provider-registry.js（去掉 OAuth/YAML/mtime）"""
from __future__ import annotations
import json
import re
from pathlib import Path
from typing import Any

from core.config import USER_DATA_DIR
from lib.providers._auth import normalize_auth_type, allows_missing_api_key
from lib.providers._validation import validate_provider_models

# 照抄 _defaultModels = JSON.parse(fs.readFileSync(fromRoot("lib","default-models.json")))
_DEFAULT_MODELS: dict[str, list] = json.loads(
    (Path(__file__).parent.parent / "default-models.json").read_text(encoding="utf-8")
)

# 运行时内部键，保存前需清除（照抄 PROVIDER_RUNTIME_META_KEYS）
_RUNTIME_META_KEYS = {"_config_error"}
_MALFORMED = "malformed_provider_config"
_INVALID_MODELS = "invalid_models_config"


def _normalize_provider_config(value: Any) -> dict:
    """照抄 normalizeProviderUserConfig"""
    if not isinstance(value, dict):
        return {"_config_error": _MALFORMED}
    next_ = dict(value)
    if "models" in next_:
        if not isinstance(next_["models"], list):
            del next_["models"]
            next_.setdefault("_config_error", _INVALID_MODELS)
        else:
            clean = []
            for m in next_["models"]:
                if isinstance(m, str) and m.strip():
                    clean.append(m.strip())
                elif isinstance(m, dict) and isinstance(m.get("id"), str) and m["id"].strip():
                    clean.append({**m, "id": m["id"].strip()})
                else:
                    next_.setdefault("_config_error", _INVALID_MODELS)
            next_["models"] = clean
    return next_


def _strip_runtime_meta(config: dict) -> dict:
    """照抄 stripProviderRuntimeMeta"""
    return {k: v for k, v in _normalize_provider_config(config).items() if k not in _RUNTIME_META_KEYS}


class ProviderRegistry:
    """照抄 provider-registry.js ProviderRegistry 类（去掉 OAuth、YAML、mtime 缓存）"""

    def __init__(self, data_dir: Path) -> None:
        self._data_dir = data_dir
        # 照抄 this._plugins = new Map()
        self._plugins: dict[str, dict] = {}
        # 照抄 this._entries = new Map()
        self._entries: dict[str, dict] = {}
        # 注册所有内置插件（照抄 for (const plugin of BUILTIN_PLUGINS)）
        self._register_builtin_plugins()

    def _register_builtin_plugins(self) -> None:
        """从 lib/providers/*.py 导入所有 PROVIDER 声明并注册"""
        from lib.providers import (
            anthropic, openai, gemini, xai, mistral, groq, perplexity,
            fireworks, together, openrouter, dashscope, deepseek, moonshot,
            zhipu, volcengine, siliconflow, baichuan, hunyuan, minimax,
            baidu_cloud, stepfun, modelscope, infini, mimo, ollama, custom,
        )
        for mod in [
            anthropic, openai, gemini, xai, mistral, groq, perplexity,
            fireworks, together, openrouter, dashscope, deepseek, moonshot,
            zhipu, volcengine, siliconflow, baichuan, hunyuan, minimax,
            baidu_cloud, stepfun, modelscope, infini, mimo, ollama, custom,
        ]:
            p = mod.PROVIDER
            self._plugins[p["id"]] = p

    # ── 用户配置 读/写 ─────────────────────────────────────────────────
    # 照抄 _loadAddedModels / _saveAddedModels（JSON 替代 YAML，去掉 mtime 缓存）

    def _load_user_providers(self) -> dict[str, dict]:
        """照抄 _loadAddedModels"""
        path = self._data_dir / "config.json"
        try:
            raw = json.loads(path.read_text(encoding="utf-8"))
            providers = raw.get("providers", {})
            if not isinstance(providers, dict):
                return {}
            return {pid: _normalize_provider_config(cfg) for pid, cfg in providers.items() if pid}
        except Exception:
            return {}

    def _save_user_providers(self, providers: dict[str, dict]) -> None:
        """照抄 _saveAddedModels（JSON + tmp→rename 原子写入）"""
        path = self._data_dir / "config.json"
        try:
            existing = json.loads(path.read_text(encoding="utf-8"))
        except Exception:
            existing = {}
        # 清除运行时内部键再写入（照抄 stripProviderRuntimeMetaMap）
        existing["providers"] = {pid: _strip_runtime_meta(cfg) for pid, cfg in providers.items()}
        tmp = path.with_suffix(".json.tmp")
        tmp.write_text(json.dumps(existing, indent=2, ensure_ascii=False), encoding="utf-8")
        tmp.replace(path)   # 原子替换（照抄 fs.renameSync）
        self._entries.clear()

    # ── reload / merge ──────────────────────────────────────────────────
    # 照抄 reload() / _merge()

    def reload(self) -> None:
        """照抄 reload()"""
        self._entries.clear()
        user = self._load_user_providers()

        # 1. 先处理所有已注册插件（照抄 for (const [id, plugin] of this._plugins)）
        for pid, plugin in self._plugins.items():
            uc = user.get(pid, {})
            self._entries[pid] = self._merge(plugin, uc, is_builtin=True)

        # 2. user config 中有但没有对应插件的（用户自定义，照抄第二段循环）
        for pid, uc in user.items():
            if pid in self._entries:
                continue
            synthetic = {
                "id": pid,
                "label": uc.get("display_name", pid),
                "auth_type": normalize_auth_type(uc.get("auth_type", "api-key")),
                "base_url": uc.get("base_url", ""),
                "api": uc.get("api", "openai-completions"),
            }
            self._entries[pid] = self._merge(synthetic, uc, is_builtin=False)

    def _merge(self, plugin: dict, user_config: dict, *, is_builtin: bool) -> dict:
        """照抄 _merge()"""
        return {
            "id": plugin["id"],
            "label": user_config.get("display_name") or plugin.get("label", plugin["id"]),
            "auth_type": normalize_auth_type(user_config.get("auth_type") or plugin.get("auth_type", "api-key")),
            "base_url": user_config.get("base_url") or plugin.get("base_url", ""),
            "api": user_config.get("api") or plugin.get("api", "openai-completions"),
            "is_builtin": is_builtin,
        }

    def get_all(self) -> dict[str, dict]:
        """照抄 getAll()"""
        if not self._entries:
            self.reload()
        return self._entries

    def get(self, provider_id: str) -> dict | None:
        """照抄 get()"""
        if not self._entries:
            self.reload()
        return self._entries.get(provider_id)

    # ── ProviderSummary（照抄 GET /providers/summary 的合并逻辑）──────────

    def get_all_summary(self) -> dict[str, dict]:
        """
        照抄 providers.js GET /providers/summary 的核心逻辑（去掉 OAuth 分支）。
        返回 { provider_id: ProviderSummary }
        """
        user = self._load_user_providers()
        result: dict[str, dict] = {}

        # 1. 先处理 user config 中出现的 provider（保持用户配置顺序）
        for name, up in user.items():
            entry = self.get(name)
            auth_type = normalize_auth_type(up.get("auth_type") or (entry["auth_type"] if entry else "api-key"))
            base_url = up.get("base_url") or (entry["base_url"] if entry else "")
            api = up.get("api") or (entry["api"] if entry else "openai-completions")
            api_key = up.get("api_key", "")
            models = up.get("models", [])
            config_error = up.get("_config_error")

            has_creds = bool(api_key or allows_missing_api_key(auth_type, base_url))
            missing: list[str] = []
            if not base_url: missing.append("base_url")
            if not has_creds: missing.append("api_key")
            if not models: missing.append("models")

            result[name] = {
                "type": "api-key",
                "auth_type": auth_type,
                "display_name": entry["label"] if entry else up.get("display_name", name),
                "base_url": base_url,
                "api": api,
                "api_key": api_key,
                "models": models,
                "custom_models": [],
                "has_credentials": has_creds,
                "supports_oauth": False,
                "can_delete": not (entry and entry["is_builtin"]),
                "config_status": "invalid" if config_error else ("ok" if not missing else "needs_setup"),
                "config_error": config_error,
                "missing_fields": missing,
            }

        # 2. PROVIDER_CATALOG 中有但 user config 里没有的（未配置状态）
        for pid, entry in self.get_all().items():
            if pid in result:
                continue
            auth_type = entry["auth_type"]
            result[pid] = {
                "type": "api-key",
                "auth_type": auth_type,
                "display_name": entry["label"],
                "base_url": entry["base_url"],
                "api": entry["api"],
                "api_key": "",
                "models": [],
                "custom_models": [],
                "has_credentials": allows_missing_api_key(auth_type, entry["base_url"]),
                "supports_oauth": False,
                "can_delete": False,
                "config_status": "needs_setup",
                "config_error": None,
                "missing_fields": ([] if allows_missing_api_key(auth_type, entry["base_url"]) else ["api_key"]) + ["models"],
            }

        return result

    # ── credentials ──────────────────────────────────────────────────────
    # 照抄 getCredentials()（去掉 OAuth 分支）

    def get_credentials(self, provider_id: str) -> dict | None:
        """照抄 getCredentials()（去掉 OAuth）"""
        user = self._load_user_providers()
        uc = user.get(provider_id)
        entry = self.get(provider_id)
        if not uc and not (entry and allows_missing_api_key(entry["auth_type"], entry.get("base_url", ""))):
            return None
        uc = uc or {}
        return {
            "api_key": uc.get("api_key", ""),
            "base_url": uc.get("base_url") or (entry["base_url"] if entry else ""),
            "api": uc.get("api") or (entry["api"] if entry else "openai-completions"),
        }

    # ── model CRUD ────────────────────────────────────────────────────────
    # 照抄 getDefaultModels / saveProvider / updateModelEntry / remove

    def get_default_models(self, provider_id: str) -> list:
        """照抄 getDefaultModels()"""
        return _DEFAULT_MODELS.get(provider_id, [])

    def save_provider(self, provider_id: str, data: dict, seed_default_models: bool = False) -> None:
        """照抄 saveProvider()"""
        user = self._load_user_providers()
        provider_data = {k: v for k, v in data.items() if k != "seed_default_models"}
        next_provider = {**(user.get(provider_id) or {}), **provider_data}

        if seed_default_models and not next_provider.get("models"):
            defaults = self.get_default_models(provider_id)
            if defaults:
                next_provider["models"] = list(defaults)

        validate_provider_models(provider_id, next_provider.get("models", []),
                                  next_provider.get("base_url", ""))
        user[provider_id] = next_provider
        self._save_user_providers(user)

    def update_model_entry(self, provider_id: str, model_id: str, meta: dict) -> None:
        """照抄 updateModelEntry()"""
        user = self._load_user_providers()
        if provider_id not in user:
            user[provider_id] = {}
        uc = user[provider_id]
        if not isinstance(uc.get("models"), list):
            uc["models"] = []

        # 允许字段白名单（照抄 ALLOWED）
        ALLOWED = {"name", "context", "maxOutput", "image", "video", "reasoning", "type"}
        safe = {k: v for k, v in meta.items() if k in ALLOWED}

        found = False
        next_models = []
        for m in uc["models"]:
            mid = m["id"] if isinstance(m, dict) else m
            if mid != model_id:
                next_models.append(m)
                continue
            found = True
            base = m if isinstance(m, dict) else {"id": mid}
            merged = {**base, **safe}
            if not merged.get("name"):
                merged.pop("name", None)
            next_models.append(merged)

        # upsert（照抄 if (!found) nextModels.push(...)）
        if not found:
            next_models.append({"id": model_id, **safe})

        validate_provider_models(provider_id, next_models, uc.get("base_url", ""))
        uc["models"] = next_models
        user[provider_id] = uc
        self._save_user_providers(user)

    def remove_provider(self, provider_id: str) -> None:
        """照抄 remove()"""
        user = self._load_user_providers()
        if provider_id not in user:
            return
        del user[provider_id]
        self._save_user_providers(user)
        # 如果有内置插件声明，以默认值重建 entry（照抄 if (this._plugins.has(providerId))）
        if provider_id in self._plugins:
            plugin = self._plugins[provider_id]
            self._entries[provider_id] = self._merge(plugin, {}, is_builtin=True)
        else:
            self._entries.pop(provider_id, None)
```

# ── models-cache 读写（照抄 readModelsCache / writeModelsCache）──────

def _cache_path() -> Path:

def _read_cache() -> dict:
    try:
        return json.loads(_cache_path().read_text(encoding="utf-8"))

def _write_cache(cache: dict) -> None:
    """照抄原子写入：tmp + rename"""
    target = _cache_path()
    tmp = target.with_suffix(".json.tmp")
    tmp.write_text(json.dumps(cache, indent=2, ensure_ascii=False), encoding="utf-8")

def _save_to_cache(provider_name: str, models: list) -> None:

    if not provider_name or not models:
        return
    from datetime import datetime, UTC
    try:
        cache = _read_cache()
        cache[provider_name] = {"models": models, "fetchedAt": datetime.now(UTC).isoformat()}
        _write_cache(cache)
    except Exception:

# ── 模型列表规范化（照抄 normalizeRemoteModels）──────────────────────

def _normalize_remote_models(data: dict, api: str) -> list:
    if api == "anthropic-messages":

             "context": m.get("max_input_tokens"), "maxOutput": m.get("max_tokens")}
            for m in data.get("data", [])
        ]
    if api == "google-generative-ai":
        result = []
        for m in data.get("models", []):
            mid = m.get("baseModelId") or (m.get("name", "")).replace("models/", "")
            if mid:
                result.append({"id": mid, "name": m.get("displayName", mid),
                               "context": m.get("inputTokenLimit"), "maxOutput": m.get("outputTokenLimit")})
        return result
    # openai-completions 默认
    return [
        {"id": m["id"], "name": m["id"],
         "context": m.get("context_length") or m.get("context_window") or m.get("max_context_length"),
         "maxOutput": m.get("max_completion_tokens") or m.get("max_output_tokens")}
        for m in data.get("data", [])
    ]

def _registry_or_defaults_fallback(name: str) -> dict:
    """照抄 registryOrDefaultsFallback()"""
    if not name:
        return {"error": "name is required for fallback", "models": []}
    defaults = get_provider_registry().get_default_models(name)
    if defaults:
        models = [{"id": mid, "name": mid, "context": None, "maxOutput": None} for mid in defaults]
        filtered, _ = filter_discovered_models(name, models)
        _save_to_cache(name, filtered)
        return {"source": "builtin", "models": filtered}
    return {"error": f'No models found for provider "{name}"', "models": []}

class FetchModelsRequest(BaseModel):
    name: str | None = None
    base_url: str | None = None
    api: str | None = None
    api_key: str | None = None

@router.post("/providers/fetch-models")
async def fetch_models(body: FetchModelsRequest):
    """照抄 POST /providers/fetch-models"""
    if not body.name and not body.base_url:
        return JSONResponse({"error": "name or base_url is required"}, status_code=400)

    # 凭证优先级：body > saved（照抄）
    saved = get_provider_registry().get_credentials(body.name) if body.name else {}
    saved = saved or {}
    effective_key = body.api_key or saved.get("api_key", "")
    effective_url = (body.base_url or saved.get("base_url", "")).rstrip("/")
    effective_api = body.api or saved.get("api", "")

    if effective_url:
        try:
            if effective_api == "anthropic-messages":
                url = f"{effective_url}/v1/models?limit=1000"
            elif effective_api == "google-generative-ai":
                url = f"https://generativelanguage.googleapis.com/v1beta/models?key={effective_key}"
            else:
                url = f"{effective_url}/models"

            headers: dict = {}
            if effective_key and effective_api:
                headers = build_auth_headers(effective_api, effective_key, allow_missing=True)

            async with httpx.AsyncClient(timeout=15) as client:
                resp = await client.get(url, headers=headers)

            # 401/403：直接返回错误，不 fallback（照抄）
            if resp.status_code in (401, 403):
                return {"error": f"HTTP {resp.status_code}: {resp.reason_phrase}", "models": []}

            if resp.is_success:
                data = resp.json()
                remote = _normalize_remote_models(data, effective_api)
                filtered, ignored = filter_discovered_models(body.name or "", remote, effective_url)
                if not filtered and ignored:
                    return {"error": f"Remote only returned invalid model ids: {ignored}", "models": [], "ignoredModels": ignored}
                _save_to_cache(body.name, filtered)
                result = {"models": filtered}
                if ignored: result["ignoredModels"] = ignored
                return result
        except Exception:
            pass  # 网络错误 → fallback

    return _registry_or_defaults_fallback(body.name or "")

## core/config.py 修改：save_user_config

照抄 `_saveAddedModels` 的合并语义，加 providers 深合并：

```python
def save_user_config(data: dict[str, Any]) -> dict[str, Any]:
    _ensure_user_data_dir()
    config_path = USER_DATA_DIR / "config.json"
    existing = load_user_config()

    # 顶层标量字段浅合并（过滤空值，不动 providers）
    flat = {
        k: v for k, v in data.items()
        if k != "providers"
        and v not in (None, "")
        and not (isinstance(v, str) and not v.strip())
    }
    existing.update(flat)

    # providers 深合并（照抄 _saveAddedModels 的合并语义）
    # value 为 None → 删除该供应商（照抄 remove()）
    # value 为 dict → 字段级合并（照抄 saveProvider() 的 partial update）
    if "providers" in data and isinstance(data["providers"], dict):
        ep = existing.get("providers") or {}
        for pid, pval in data["providers"].items():
            if pval is None:
                ep.pop(pid, None)
            elif isinstance(pval, dict):
                ep[pid] = {**(ep.get(pid) or {}), **{k: v for k, v in pval.items() if v is not None}}
        existing["providers"] = ep

    tmp = config_path.with_suffix(".json.tmp")
    tmp.write_text(json.dumps(existing, indent=2, ensure_ascii=False), encoding="utf-8")
    tmp.replace(config_path)   # 原子写入
    return existing
```

### ApiKeyCredentials.tsx：getApiKeySavePlan 内联

openhanako 的 `getApiKeySavePlan` 判断是否要验证、是否要 seed 模型。Lumen 内联逻辑：

   - 若 `keyEdited` 或 `isPresetSetup`：调 `/api/providers/test` 验证
   - 验证通过后构造 payload：`{ api_key: keyVal, base_url: urlVal }`

   - PUT `/api/config` 保存 `{ providers: { [providerId]: payload } }`
2. Base URL `onBlur`：自动 PUT，不验证
3. API 类型 `onChange`：自动 PUT，不验证
